function dbg_log(x)
{
 check(is_obj,x)

 const stack=dbg_callstack(x)
 const trace=dbg_backtrace(x)
 const origin=dbg_origin(x)
 
 tbl_align_r(stack,"fn")
 tbl_align_r(trace,"fn")
 
 if(is_full(origin))
 {
  const s=tbl_render(origin)

  log(s)
 }
 
 if(is_empty(stack))
 {
  const s=tbl_render(trace)
  
  log(s)
 }
 else
 {
  const s=tbl_render(stack)
  
  log(s)
 }
}

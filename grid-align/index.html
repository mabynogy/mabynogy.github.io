fn grid_align x y:spread
 check is_obj x
 
 if same y.length 1
  let y_ front y
  let y y_

  check is_str y
  
  forof x.columns
   grid_align x v y
  end
 elseif same y.length 2
  let y_ at y 0
  let z at y 1
  let y y_
  
  check is_cool y
  check is_str z
 
  let n grid_index x y

  forof x.table.children
   let td at v.children n
   
   dom_align td z
  end
 else
  stop
end

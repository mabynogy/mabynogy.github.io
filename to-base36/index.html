fn to_base36 x
 check is_cool x

 if is_uint x
  ret x.toString 36
 elseif is_str x
  var r ""
  
  forof x
   if is_single v
    let n asc v
    let s to_base36 n
    
    assign r concat r s 
   elseif is_pair v
    let n1 v.charCodeAt 0
    let n2 v.charCodeAt 1
    let s1 to_base36 n1
    let s2 to_base36 n2

    assign r concat r s1 s2 
   else
    stop
  end
  
  ret r
 else
  stop
end

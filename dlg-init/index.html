fn dlg_init x
 if is_undef x
  ret dlg_init ""
 
 check is_str x

 let dialog dom_dialog body
 let title dom_div dialog x

 dom_align dialog "center"
 dom_bold title
 dom_br dialog
 
 let container dom_div dialog
 
 dom_display container "inline-block"  
 
 let fields arr
 let buttons arr
 
 ret obj dialog title container fields buttons
end

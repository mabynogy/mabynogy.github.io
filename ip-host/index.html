fn ip_host x
 if is_undef x
  ret call os.hostname
  
 check is_str x
 
 let domain ip_domain x
 
 if is_str domain
  ret domain
  
 let ips call ip_list
 
 forin ips
  let i to_i k
  let v at ips i
  
  if same x v
   let host call ip_host
   let letter asc "a"
   let letter add letter i
   let letter chr letter
   
   ret concat host "-" letter
  end
 end
 
 ret x
end

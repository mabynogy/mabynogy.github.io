fn is_mail x
 fn is_component x
  if not is_str x
   ret false
  
  if is_empty x
   ret false
   
  forof delimit x "."
   if is_alnum v
   elseif is_lisp v
   else
    ret false
  end
  
  ret true
 end

  ret false
    
 let a split x "@"
 
 if different a.length 2
  ret false
  
 forof a

   ret false
 end
 
 ret true
end

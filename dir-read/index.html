function dir_read(x,y)
{
 check(is_str,x)
 
 if(is_undef(y))
  return dir_read(x,is_file)
  
 check(is_fn,y)
 
 const r=arr()
 const dir=path_real(x)
 
 for(const v of fs.readdirSync(dir))
 {
  const path=path_concat(dir,v)
  const b=y(path)
  
  if(is_true(b))
   push(r,path)
  else
   check(is_false,b)
 }
 
 return r
}

fn ip_domain x
 check is_str x
 
 let timeout 2
 let parameters arr "host" "-W" timeout

 if is_ip4 x
  push parameters "-4"
 elseif is_ip6 x
  push parameters "-6"
 else
  stop
  
 push parameters x
   
 let r os_execute parameters:spread
 let r strip_r r "."
 let a split r " "
 let r back r
 
 if not is_domain r
  ret null
 
 ret r
end

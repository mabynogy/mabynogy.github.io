fn pms_page_admin_links x
 check is_obj x

 fn on_enter x
  check is_obj x
  
  forof x.children
   dom_visible v "visible"
  end
 end

 fn on_click x
  check is_obj x
  
  on_enter x
 end
 
 fn on_link x y
  check is_obj x
  check is_obj y

  let edit concat "https://admin.pimaastore.com/?q=p," y.id
  let mobile concat "/?n=" y.id "&v=m"
  let desktop concat "/?n=" y.id "&v=d"
  let o obj edit mobile desktop
  let url get o x.value

  open url "_blank"
 end

 let div dom_div body
 let choose "<>"
 let edit "Modifier"
 let mobile "Mobile"
 let desktop "Desktop"
 let o obj choose edit mobile desktop
 let select dom_select div o

 dom_padding_top div padding
 dom_visible select "hidden"

 assign div.onmouseenter partial on_enter div
 assign div.onclick partial on_click div
 
 assign select.onchange partial on_link select x
end

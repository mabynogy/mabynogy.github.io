fn sitemap_index x
 check is_obj x
 
 let html html_init "html"
 let head html_init "head"
 let meta html_init "meta"
 let title html_init "title"
 let body html_init "body"
 let script html_init "script"

 html_attribute meta "charset" "utf-8"

 html_text title "doc"

 html_attribute script "src" "/spa-mabynogy.js"
 
 html_add body script
 
 html_add head meta
 html_add head title
 
 html_add html head
 html_add html body
 
 let s html_render html
 let path path_concat x.tmp "/index.html"
 
 file_save path s
end

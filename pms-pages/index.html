 fn get_footer x
  check is_arr x
  
  let r arr
  let ids arr 10 18 28
  
  forof x
   if contain ids v.id
    push r v
  end
  
  ret r
 end

 let r run ui_page_list
 let r to_flat r
 
 forof r
  assign v.content to_html v.content
  assign v.url concat "/?n=" v.id
  assign v.footer get_footer r
 end
 
 let r to_tree r
 
 ret r

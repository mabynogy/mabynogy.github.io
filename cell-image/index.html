fn cell_image x
 check is_def x
 
 if not is_obj x
  ret null 
  
 if not has x "is_image"
  ret null
  
 let is_image x.is_image
  
 if not is_image
  ret null
  
 let url x.url       
 let name x.visible_name
 let r dom_a null "" url true
 let img dom_img r url
 
 dom_class r "image-link"
 
 if is_str name
  assign img.title name
 
 ret r
end

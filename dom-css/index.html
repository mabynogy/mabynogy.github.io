fn dom_css x y z a:spread
 check is_str x
 check is_str y
 
 if is_num z
  let s to_str z
  
  ret dom_css x y s a:spread
 end

 check is_str z
 
 let style dom_by_tag document "style"
 let style front style
 
 let content dom_text style
 let css concat y ":" z ";"
 let css brace css
 let css concat x css "\n"
 let css concat content css
 
 dom_text style css
 
 if is_full a
  dom_css x a:spread
end

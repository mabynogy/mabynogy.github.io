fn spa_compile x y
 check is_str x
 check is_str y
 
 let env call env_init
 let index path_concat env.prg x x
 let index concat index ".cs"
 let html path_concat env.tmp x
 let html concat html ".html"
 let home path_concat y "index.html"
 
 os_system "prg-compile" index "--html"
 
 if not is_file html
  ret
 
 os_system "mv" "--force" html home
end

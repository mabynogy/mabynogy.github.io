gn ui_page_down x
 check is_obj x
 
 gn get_after x
  check is_obj x
  
  let siblings run ui_page_siblings x
  
  while is_full siblings
   let page shift siblings
   
   if same page.id x.id
    if is_empty siblings
     ret null
    
    ret front siblings
   end
  end
  
  stop
 end

 let after run get_after x
 
 if is_null after
  ret

 let pages run ui_page_find
 let a arr
 
 while is_full pages
  let page shift pages

  if same page.id x.id
  elseif same page.id after.id
   push a after
   push a x
  else
   push a page
 end

 run ui_page_reorder a 
end

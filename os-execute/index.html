fn os_execute x y:spread
 check is_str x

 let shell true
 let maxBuffer mul 1 1024 1024 1024
 let encoding "utf8"
 let option obj shell maxBuffer encoding
 let streams cp.spawnSync x y option
 let out to_str streams.stdout
 let err to_str streams.stderr
 let r concat out "\n" err
 let r trim_r r

 ret r
end

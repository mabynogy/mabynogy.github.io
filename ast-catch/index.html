function ast_catch(x,y)
{
 check(is_arr,x)
 check(is_arr,y)

 const block=ast_block(y)
 
 if(is_empty(x))
  return concat("catch\n",block)
 
 check(is_single,x)
 
 const identifier=front(x)
 
 check(is_identifier,identifier)
 
 const list=paren(identifier)
 
 return concat("catch",list,"\n",block)
}

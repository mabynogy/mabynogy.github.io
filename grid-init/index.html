fn grid_init x y:spread  
 check is_obj x
 
 let columns arr
 let titles y
 let table dom_table x
 let tr dom_tr table
 let headers obj
 let tooltip null
 
 forof y
  let column to_lower v  
  var key column
  let th dom_th tr
  let div dom_div th v
  
  if same key "#"
   assign key "sharp"

  push columns key
  put headers key th
  
  dom_padding th padding
  dom_border th border  
 end

 assign r obj columns titles table headers tooltip

 grid_init_sort r
 
 ret r
end

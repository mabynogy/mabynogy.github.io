fn get_user_agent
 let agent navigator.userAgent
 let agent to_lower agent
 
 if contain agent "googlebot"
  ret "googlebot"
 elseif contain agent "android"
  ret "android"
 elseif contain agent "iphone"
  ret "iphone"
 elseif contain agent "firefox"
  ret "firefox"
 elseif contain agent "edge"
  ret "edge"
 elseif contain agent "chrome"
  ret "chrome"
 elseif contain agent "safari"
  ret "safari"
 else
  ret "unknown"
end
